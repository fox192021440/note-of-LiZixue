#### 批量添加文件序号命令


```
$i=1; Get-ChildItem -File | ForEach-Object { Rename-Item $_ -NewName ("{0}-{1}{2}" -f $i, $_.BaseName, $_.Extension); $i++ }
```

#### 批量添加文件夹序号命令

```
$i=1; Get-ChildItem -Directory | ForEach-Object { Rename-Item $_ -NewName ("{0}-{1}" -f $i, $_.Name); $i++ }
```

#### 批量修改文件命名命令
```
# 固定字段配置
$projectName = "WorkSpace"
$fileType = "Project&Plan"
$dateTimeFormat = "yyyyMMdd-HHmmss"  # 时间格式：年月日-时分秒（精确到秒）
$startNum = 1                        # 起始序号
$numDigit = 3                        # 序号补零位数（如3→001）

# 按创建时间排序并批量重命名
Get-ChildItem -File | Sort-Object CreationTime | ForEach-Object {
    $createDateTime = $_.CreationTime.ToString($dateTimeFormat)
    $seqNum = $startNum.ToString("D$numDigit")
    $newName = "{0}-{1}-{2}-{3}{4}" -f $projectName, $fileType, $createDateTime, $seqNum, $_.Extension
    Rename-Item $_ -NewName $newName
    Write-Host "已重命名：$($_.Name) → $newName"
    $startNum++
}

Write-Host "`n所有文件处理完成！"

```

#### 批量修改文件命名命令（仅修改未按照模板修改命名的文件）

```
<#
.SYNOPSIS
批量重命名未按指定模板命名的文件，已按模板命名的文件保持不变。

.DESCRIPTION
该脚本会识别并处理未按照"WorkSpace-Project&Plan-yyyyMMdd-HHmmss-XXX.扩展名"格式命名的文件，
已按此格式命名的文件将被自动跳过，不会被修改。
#>

# 配置参数 - 可根据需要修改
$projectName = "WorkSpace"               # 项目名称
$fileType = "Project&Plan"               # 文件类型
$dateTimeFormat = "yyyyMMdd-HHmmss"      # 日期时间格式（精确到秒）
$startNum = 1                            # 未命名文件的起始序号
$numDigit = 3                            # 序号位数（补零）

# 正则表达式模式 - 用于匹配已按模板命名的文件
# 格式：WorkSpace-Project&Plan-8位日期-6位时间(时分秒)-3位序号.扩展名
$pattern = "^$($projectName)-$($fileType)-\d{8}-\d{6}-\d{$numDigit}\.\w+$"

try {
    # 获取所有文件，筛选出未按模板命名的文件，并按创建时间排序
    $filesToRename = Get-ChildItem -File | 
        Where-Object { $_.Name -notmatch $pattern } | 
        Sort-Object CreationTime

    # 统计符合条件的文件数量
    $totalFiles = $filesToRename.Count
    
    if ($totalFiles -eq 0) {
        Write-Host "`n信息：所有文件均已按模板命名，无需修改。`n" -ForegroundColor Green
        exit 0
    }

    Write-Host "`n发现 $totalFiles 个未按模板命名的文件，开始处理...`n" -ForegroundColor Cyan

    # 批量重命名文件
    $currentNum = $startNum
    foreach ($file in $filesToRename) {
        # 获取文件创建时间并格式化
        $createDateTime = $file.CreationTime.ToString($dateTimeFormat)
        
        # 格式化序号（补零）
        $seqNum = $currentNum.ToString("D$numDigit")
        
        # 构建新文件名
        $newName = "{0}-{1}-{2}-{3}{4}" -f $projectName, $fileType, $createDateTime, $seqNum, $file.Extension
        
        # 执行重命名
        Rename-Item -Path $file.FullName -NewName $newName -ErrorAction Stop
        
        # 输出处理信息
        Write-Host "已重命名：$($file.Name) -> $newName" -ForegroundColor Green
        
        $currentNum++
    }

    Write-Host "`n处理完成！成功重命名 $totalFiles 个文件。`n" -ForegroundColor Green
}
catch {
    Write-Host "`n处理过程中发生错误：$($_.Exception.Message)`n" -ForegroundColor Red
    exit 1
}
    
```

### 批量添加双链

```
<#
    Obsidian批量替换概念为链接脚本（适配中文词汇）
    仅处理目标文件夹内直接的.md文件，不递归子文件夹
    路径配置：
    1. 链接库文件：E:\BaiduSyncdisk\obsidian\work\Templates\20260218-174003.md
    2. 批量处理目录：E:\BaiduSyncdisk\obsidian\work
#>
# ====================== 【路径无需修改】 ======================
$linkLibPath = "E:\BaiduSyncdisk\obsidian\work_backup\Templates\20260218-174003.md"  # 链接库文件路径
$targetDir = "E:\BaiduSyncdisk\obsidian\work_backup"                                 # 批量处理的笔记目录
# ==================================================================

# 编码设置（避免Markdown文件乱码）
$PSDefaultParameterValues['*:Encoding'] = 'utf8'

# 第一步：读取并解析链接库，提取「概念名-链接」映射
if (-not (Test-Path $linkLibPath)) {
    Write-Host "[ERROR] 链接库文件不存在：$linkLibPath" -ForegroundColor Red
    pause
    exit
}

# 正则匹配Obsidian链接格式：[[任意内容|概念名]]（兼容带锚点的链接）
$linkRegex = '\[\[(.*?)\|(.*?)\]\]'
$linkMap = @{}  # 存储「概念名-完整链接」的哈希表

# 读取链接库内容并解析（兼容老版本PowerShell）
Get-Content -Path $linkLibPath | ForEach-Object {
    $line = $_.Trim()
    if ($line -match $linkRegex) {
        $conceptName = $matches[2].Trim()
        $fullLink = "[[" + $matches[1] + "|" + $conceptName + "]]"
        $linkMap[$conceptName] = $fullLink
        Write-Host "[SUCCESS] 读取到链接：$conceptName → $fullLink" -ForegroundColor Green
    }
}

if ($linkMap.Count -eq 0) {
    Write-Host "[ERROR] 链接库中未找到有效链接，请检查格式" -ForegroundColor Red
    pause
    exit
}

# 验证：手动输出读取到的「伺服电机」「链条」链接（排查是否解析成功）
if ($linkMap.ContainsKey("伺服电机")) {
    Write-Host "`n[验证] 已读取到伺服电机链接：$($linkMap['伺服电机'])" -ForegroundColor Yellow
} else {
    Write-Host "`n[验证] 未读取到伺服电机链接，请检查链接库格式" -ForegroundColor Red
}
if ($linkMap.ContainsKey("链条")) {
    Write-Host "[验证] 已读取到链条链接：$($linkMap['链条'])" -ForegroundColor Yellow
} else {
    Write-Host "[验证] 未读取到链条链接，请检查链接库格式" -ForegroundColor Red
}

# 第二步：按「概念长度从长到短」排序（避免短概念覆盖长概念）
$sortedConcepts = $linkMap.Keys | Sort-Object -Property Length -Descending

# 第三步：批量处理目标目录下【直接的】Markdown文件
$mdFiles = Get-ChildItem -Path $targetDir -Filter "*.md"
Write-Host "`n[INFO] 共找到 $($mdFiles.Count) 个Markdown文件，开始批量替换..." -ForegroundColor Cyan

foreach ($file in $mdFiles) {
    $content = Get-Content -Path $file.FullName -Raw
    $replaceCount = 0
    $originalContent = $content

    # 遍历所有概念，逐个替换（适配中文，移除单词边界\b，优化断言）
    foreach ($concept in $sortedConcepts) {
        $link = $linkMap[$concept]
        $escapedConcept = [regex]::Escape($concept)
        
        # 【核心优化】移除\b，改用「非链接字符断言」适配中文
        # (?<!\[) 确保前面不是[，(?!\]) 确保后面不是]
        # (?<!\w) 和 (?!\w) 兼容中英文混合场景的边界
        $replaceRegex = "(?<!\[\[)(?<!\w)$escapedConcept(?!\w)(?!\]\])"
        
        # 统计匹配次数并替换
        $matches = [regex]::Matches($originalContent, $replaceRegex)
        if ($matches.Count -gt 0) {
            $content = $content -replace $replaceRegex, $link
            $replaceCount += $matches.Count
            # 调试：输出该概念的匹配情况
            Write-Host "[调试] $($file.Name) 中「$concept」匹配到 $($matches.Count) 处" -ForegroundColor DarkGray
        }
    }

    # 只有内容变化时才写入文件
    if ($content -ne $originalContent) {
        Set-Content -Path $file.FullName -Value $content
        Write-Host "[SUCCESS] $($file.Name)：替换了 $replaceCount 处内容" -ForegroundColor Green
    } else {
        Write-Host "[INFO] $($file.Name)：无需要替换的内容" -ForegroundColor Gray
    }
}

Write-Host "`n[COMPLETE] 批量替换完成！共处理 $($mdFiles.Count) 个文件" -ForegroundColor Green
pause
```
